# ── Core ML ──────────────────────────────────────────────────────────────────
torch>=2.3.0
torchvision>=0.18.0

# ── HuggingFace stack ─────────────────────────────────────────────────────────
# transformers 4.52+ has native Qwen3-VL support.
# Qwen3.5-27B (default model) uses a GatedDeltaNet + sparse MoE architecture
# with early-fusion multimodal support — requires transformers>=4.52 and
# trust_remote_code=True (set automatically by the scripts).
transformers>=4.52.0
accelerate>=0.34.0
huggingface_hub>=0.24.0
tokenizers>=0.21.0

# ── vLLM — high-throughput inference engine ──────────────────────────────────
# Handles tensor parallelism, continuous batching, PagedAttention.
# Supports Qwen3.5-27B, Qwen3-VL, Qwen2.5-VL, Llama 3.x, Mistral, and 40+
# other model families.
# vLLM 0.7+ includes Qwen3-VL; Qwen3.5-27B uses trust_remote_code for
# custom architecture layers (GatedDeltaNet).
vllm>=0.7.0

# ── Image handling ────────────────────────────────────────────────────────────
pillow>=10.3.0

# ── Progress / utilities ──────────────────────────────────────────────────────
tqdm>=4.66.0

# ─────────────────────────────────────────────────────────────────────────────
# NOTES ON MODEL-SPECIFIC PACKAGES
# ─────────────────────────────────────────────────────────────────────────────
#
# Qwen3.5-27B (default --vlm_model, unified mode):
#   No extra packages needed — transformers>=4.52 + trust_remote_code handles
#   the GatedDeltaNet layers.  Natively multimodal (early fusion) so it handles
#   both image captioning (Step 1) and text reasoning (Steps 2a/2b/3).
#
# Qwen3-VL (--vlm_model Qwen/Qwen3-VL-*):
#   No extra packages needed — transformers>=4.52 handles everything natively.
#
# Qwen2.5-VL (--vlm_model Qwen/Qwen2.5-VL-*):
#   No extra packages needed with transformers>=4.49.
#   If using the HF fallback (not vLLM), optionally install:
#     pip install qwen-vl-utils>=0.0.8
#
# Qwen2-VL (older, --vlm_model Qwen/Qwen2-VL-*):
#   pip install qwen-vl-utils>=0.0.8   # required for HF fallback
#
# AWQ quantized models (--quantization awq):
#   pip install autoawq>=0.2.5
#
# GPTQ quantized models (--quantization gptq):
#   pip install auto-gptq>=0.7.0
#
# FP8 (--quantization fp8):
#   Supported natively by vLLM on H100/Ada Lovelace; no extra package needed.
